/* Creación de usuario si está trabajando con BD Oracle XE */
CREATE USER PRUEBA_P2 IDENTIFIED BY "Prueba.parcial_2"
DEFAULT TABLESPACE "USERS"
TEMPORARY TABLESPACE "TEMP";
ALTER USER PRUEBA_P2 QUOTA UNLIMITED ON USERS;
GRANT CREATE SESSION TO PRUEBA_P2;
GRANT "RESOURCE" TO PRUEBA_P2;
ALTER USER PRUEBA_P2 DEFAULT ROLE "RESOURCE";
