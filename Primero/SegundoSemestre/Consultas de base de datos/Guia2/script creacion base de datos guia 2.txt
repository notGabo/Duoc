CREATE SEQUENCE sq_cat;

CREATE TABLE zona (
    idzona NUMBER(1) NOT NULL,
    nomzona VARCHAR2(20) NOT NULL,
    CONSTRAINT pk_zona PRIMARY KEY  (idzona)
);

CREATE TABLE comuna (
   idcomuna NUMBER(3) NOT NULL,
   nomcomuna VARCHAR2(30) NOT NULL,
   CONSTRAINT pk_comuna PRIMARY KEY (idcomuna),
   CONSTRAINT ak_nomcomuna_comuna UNIQUE (nomcomuna)
 );

CREATE TABLE categoria (
   idcategoria NUMBER(1) NOT NULL,
   nomcategoria VARCHAR2(20) NOT NULL,
   smin NUMBER(7) NOT NULL
   CONSTRAINT chk_smin CHECK (smin <= 1500000),
   smax NUMBER(7) NOT NULL,
   CONSTRAINT chk_smin_smax CHECK (smin < smax),
   CONSTRAINT pk_categoria PRIMARY KEY (idcategoria)
);

CREATE TABLE oficina (
    numoficina NUMBER(3) NOT NULL,
    diroficina VARCHAR2(60) NOT NULL,
    idcomuna NUMBER(3),
    CONSTRAINT pk_oficina PRIMARY KEY (numoficina)
);

CREATE TABLE tipo (
   idtipo VARCHAR2(1) NOT NULL,
   nomtipo VARCHAR2(15) NOT NULL,
   CONSTRAINT pk_tipo PRIMARY KEY (idtipo)
);

CREATE TABLE propietario (
   rutpro VARCHAR2(10) NOT NULL,
   paterno VARCHAR2(15) NOT NULL,
   materno VARCHAR2(15) NOT NULL,
   nombre VARCHAR2(25) NOT NULL,
   direccion VARCHAR2(60) NOT NULL,
   ecivil VARCHAR2(15)
   CONSTRAINT chk_ecivil1 CHECK (ecivil IS NULL OR 
   (ecivil IN ('Soltero', 'Casado', 'Divorciado', 'Viudo'))),
   fono1 VARCHAR2(15) NOT NULL,
   fono2 VARCHAR2(15),
   idcomuna NUMBER(3),
   CONSTRAINT pk_propietario PRIMARY KEY (rutpro)
);

CREATE TABLE cliente (
   rutcli VARCHAR2(10) NOT NULL,
   nomcli VARCHAR2(50) NOT NULL,
   direccion VARCHAR2(60) NOT NULL,
   ecivil VARCHAR2(15)
   CONSTRAINT chk_ecivil2 CHECK (ecivil IS NULL OR 
   (ecivil IN ('Soltero', 'Casado', 'Divorciado', 'Viudo'))),
   fono1 VARCHAR2(15) NOT NULL,
   fono2 VARCHAR2(15),
   rentamaxima NUMBER(7) NOT NULL,
   idcomuna NUMBER(3),
   idtipo VARCHAR2(1) NOT NULL,
   CONSTRAINT pk_cliente PRIMARY KEY (rutcli)
);

CREATE TABLE empleado (
   rutemp VARCHAR2(10) NOT NULL,
   paterno VARCHAR2(15) NOT NULL,
   materno VARCHAR2(15) NOT NULL,
   nombre VARCHAR2(25) NOT NULL,
   direccion VARCHAR2(60) NOT NULL,
   ecivil VARCHAR2(15)
   CONSTRAINT chk_ecivil3 CHECK (ecivil IS NULL OR 
   (ecivil IN ('Soltero', 'Casado', 'Divorciado', 'Viudo'))),
   sexo CHAR(1)
   CONSTRAINT chk_sexo CHECK (sexo IS NULL OR (sexo IN('F', 'M'))),
   fono1 VARCHAR2(15) NOT NULL,
   fono2 VARCHAR2(15),
   fecnac DATE,
   fecing DATE NOT NULL,
   puntaje NUMBER(3) NOT NULL,
   sueldo NUMBER(7) NOT NULL,
   idcomuna NUMBER(3),
   idcategoria NUMBER(1),
   numoficina NUMBER(3),
   idzona NUMBER(1),
   rutsup VARCHAR2(10),
   CONSTRAINT pk_empleado PRIMARY KEY (rutemp)   
);

CREATE TABLE propiedad (
   numpropiedad NUMBER(6) NOT NULL,
   direccion VARCHAR2(60) NOT NULL,
   superficie NUMBER(5,2) NOT NULL,
   dormitorios NUMBER(1),
   banos NUMBER(1),
   renta NUMBER(7) NOT NULL,
   gastocomun NUMBER(6),
   idtipo VARCHAR2(1) NOT NULL,
   idcomuna NUMBER(3) NOT NULL,
   rutpro VARCHAR2(10) NOT NULL,
   rutemp VARCHAR2(10),
   CONSTRAINT pk_propiedad PRIMARY KEY (numpropiedad)
);

CREATE TABLE visita (
   numpropiedad NUMBER(6) NOT NULL,
   rutcli VARCHAR2(10) NOT NULL,
   fecha DATE NOT NULL,
   hora VARCHAR2(8) NOT NULL,
   comentarios VARCHAR2(500),
   CONSTRAINT pk_visita PRIMARY KEY (numpropiedad, rutcli, fecha)
);

CREATE TABLE lista (
  lista VARCHAR2(10),
  punt_min NUMBER(3),
  punt_max number(3)
);

CREATE OR REPLACE TRIGGER tr_sqcat
BEFORE INSERT ON categoria
FOR EACH ROW 
WHEN (NEW.idcategoria IS NULL)
DECLARE
BEGIN 
   SELECT sq_cat.NEXTVAL INTO :NEW.idcategoria FROM DUAL;
END;
/

CREATE OR REPLACE TRIGGER tr_sqofi
BEFORE INSERT ON oficina
FOR EACH ROW
DECLARE
BEGIN
   SELECT sq_ofi.NEXTVAL INTO :NEW.numoficina FROM DUAL;   
END;
/

CREATE OR REPLACE TRIGGER tr_sqcom
BEFORE INSERT ON comuna
FOR EACH ROW
DECLARE
BEGIN
   SELECT sq_com.NEXTVAL INTO :NEW.idcomuna FROM DUAL;   
END;
/
